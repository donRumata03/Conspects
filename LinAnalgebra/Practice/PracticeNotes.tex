    \begin{multline}
            \vec{v} = c_1 \times \vec{vb_1} + c_2 \times \vec{vb_2} + c_2 \times \vec{vb_2} = \\
            c_1 \times \begin{pmatrix}
                {vb_1}_{x}\\
                {vb_2}_{y} \\
                {vb_3}_{z}
        \end{pmatrix} + 
        c_2 \times \begin{pmatrix}
            {vb_2}_{x}\\
            {vb_2}_{y} \\
            {vb_2}_{z}
        \end{pmatrix}
        + c_3 \times \begin{pmatrix}
        {vb_3}_{x}\\
        {vb_3}_{y} \\
        {vb_3}_{z}
        \end{pmatrix} = \begin{pmatrix}
            {v}_{x}\\
            {v}_{y} \\
            {v}_{z}
            \end{pmatrix}
    \end{multline}
